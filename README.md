## Install

```bash
npm install --save material-state-component
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'material-state-component'
import 'material-state-component/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
